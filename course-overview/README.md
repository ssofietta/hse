$$
\begin{aligned}
𝑂_{𝑓𝑖𝑛𝑎𝑙} &= 0.03⋅(𝑂_1+𝑂_2+𝑂_3)+0.05⋅(𝑂_4+𝑂_5) +\\
&+0.035⋅(𝑂_6+𝑂_7) + 0.01⋅𝑂_8+ 0.29⋅𝑂_9 +\\
&+ 0.29⋅𝑂_{10} +0.15⋅𝑂_{11}
\end{aligned}
$$

{% endhint %}
